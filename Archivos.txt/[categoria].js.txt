import ProductCard from '../../components/Product/ProductCard';
import Link from 'next/link';
import Head from 'next/head';

export async function getServerSideProps({ params }) {
  const slug = params.categoria.toLowerCase().trim(); // "carteras", "collares-perlas", etc.

  try {
    const res = await fetch(`http://localhost:5000/api/v1/productos`);
    const data = await res.json();
    const allProducts = data.products || [];

    // LOGS DE DEPURACIÓN (los verás en la terminal del frontend)
    console.log("Slug recibido en [categoria].js:", slug);
    console.log("Categorías encontradas en la hoja:", [...new Set(allProducts.map(p => p.Categoría))]);

    const productos = allProducts.filter(product => {
      if (!product.Categoría) return false;

      const categoriaLimpia = product.Categoría
        .toString()
        .toLowerCase()
        .trim()
        .replace(/\s+/g, '-')        // espacios → guion
        .replace(/[^a-z0-9-]/g, '')  // quita acentos, ñ, etc.
        .replace(/-+/g, '-');

      const coincide = categoriaLimpia === slug;

      if (coincide) {
        console.log(`MATCH → SKU: ${product.ID_SKU} | Categoría original: "${product.Categoría}" → slug: "${categoriaLimpia}"`);
      }

      return coincide;
    });

    console.log(`Productos encontrados para "${slug}": ${productos.length} (de ${allProducts.length} totales)`);

    const titulo = productos.length > 0
      ? productos[0].Categoría.trim()
      : slug.charAt(0).toUpperCase() + slug.slice(1).replace(/-/g, ' ');

    return {
      props: {
        products: productos,
        titulo,
      },
    };
  } catch (error) {
    console.error("ERROR CRÍTICO en getServerSideProps de categoría:", error);
    return {
      props: {
        products: [],
        titulo: "Categoría no encontrada",
      },
    };
  }
}

export default function Categoria({ products, titulo }) {
  console.log("Products recibidos en el componente:", products); // ← AGREGÁ ESTE LOG

  return (
    <>
      <Head>
        <title>{titulo} • Roxycamval Joyería</title>
        <meta name="description" content={`Explora nuestra colección exclusiva de ${titulo.toLowerCase()}`} />
      </Head>

      <div className="min-h-screen bg-gradient-to-b from-black via-gray-900 to-black text-white py-20">
        <div className="container mx-auto px-6 max-w-7xl">

          <Link href="/" className="inline-block mb-10 text-pink-400 hover:text-pink-300 font-bold text-lg">
            ← Volver al inicio
          </Link>

          <h1 className="text-6xl md:text-8xl font-black text-center mb-20 text-transparent bg-clip-text bg-gradient-to-r from-pink-500 to-purple-600 drop-shadow-2xl">
            {titulo}
          </h1>

          {/* LOG PARA VER EN CONSOLA DEL NAVEGADOR */}
          {console.log("Renderizando productos:", products)}

          {products.length === 0 ? (
            <div className="text-center py-32">
              <p className="text-4xl text-gray-400 mb-8">
                Pronto más productos en esta categoría
              </p>
            </div>
          ) : (
            <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-8 lg:gap-12">
              {products.map(p => (
                <ProductCard key={p.ID_SKU} product={p} />
              ))}
            </div>
          )}
        </div>
      </div>
    </>
  );
}
